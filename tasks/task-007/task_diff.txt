+    const { n } = req.body;
+
+    if (n === undefined) {

+      return res.status(400).json({ error: 'n must be non-negative' });

+    const fibonacci = [];
+    for (let i = 0; i < count; i++) {
+      if (i === 0) {
+        fibonacci.push(0);
+      } else if (i === 1) {
+        fibonacci.push(1);
+      } else {
+        fibonacci.push(fibonacci[i-1] + fibonacci[i-2]);
+      }
+    }
+

+// Calculate factorial
+router.post('/factorial', auth, (req, res) => {
+  try {
+    const { number } = req.body;
+
+    if (number === undefined) {
+      return res.status(400).json({ error: 'number is required' });
+    }
+
+    const num = parseInt(number);
+    if (num < 0) {
+      return res.status(400).json({ error: 'number must be non-negative' });
+    }
+
+    const factorial = (n) => {
+      if (n === 0 || n === 1) return 1;
+      return n * factorial(n - 1);
+    };
+
+    res.json({ result: factorial(num) });
+  } catch (error) {
+    res.status(500).json({ error: 'Failed to calculate factorial' });
+  }

+
